let currentDeck = 0;
let currentCard = 0;
let showingAnswer = false;
let filteredIndexes = [];

function cardKey(type, deckIndex, cardIndex) {
return `nessus_ot_${type}_${deckIndex}_${cardIndex}`;
}

function isKnown(deckIndex, cardIndex) {
return localStorage.getItem(cardKey("known", deckIndex, cardIndex)) === "true";
}

function isStarred(deckIndex, cardIndex) {
return localStorage.getItem(cardKey("starred", deckIndex, cardIndex)) === "true";
}

function buildDeckList() {
const list = document.getElementById("deckList");
list.innerHTML = "";

decks.forEach((deck, index) => {
const known = deck.cards.filter((_, i) => isKnown(index, i)).length;
const btn = document.createElement("button");
btn.className = "deck-btn" + (index === currentDeck ? " active" : "");
btn.innerHTML = `<strong>${deck.title}</strong><br><span class="small">${known}/${deck.cards.length} known</span>`;
btn.onclick = () => {
currentDeck = index;
currentCard = 0;
showingAnswer = false;
refresh();
};
list.appendChild(btn);
});
}

function applyFilter() {
const mode = document.getElementById("mode").value;
const search = document.getElementById("searchBox").value.toLowerCase().trim();

filteredIndexes = decks[currentDeck].cards
.map((_, index) => index)
.filter(index => {
const card = decks[currentDeck].cards[index];
const text = (card[0] + " " + card[1]).toLowerCase();

const searchMatch = !search || text.includes(search);
const modeMatch =
mode === "all" ||
(mode === "unknown" && !isKnown(currentDeck, index)) ||
(mode === "starred" && isStarred(currentDeck, index));

return searchMatch && modeMatch;
});

if (currentCard >= filteredIndexes.length) {
currentCard = Math.max(0, filteredIndexes.length - 1);
}
}

const list = document.getElementById("cardList");
list.innerHTML = "";

filteredIndexes.forEach((realIndex, displayIndex) => {
const card = decks[currentDeck].cards[realIndex];
const div = document.createElement("div");
div.className = "mini-card";
div.innerHTML = `
<strong>${displayIndex + 1}. ${card[0]}</strong>
<div class="small">
${isKnown(currentDeck, realIndex) ? "✅ Known" : "❌ Unknown"}
${isStarred(currentDeck, realIndex) ? "⭐ Starred" : ""}
</div>
`;
div.onclick = () => {
currentCard = displayIndex;
showingAnswer = false;
render();
};
list.appendChild(div);
});

if (filteredIndexes.length === 0) {
list.innerHTML = `<p class="small">No cards match this filter.</p>`;
}
}

function renderProgress() {
const total = decks[currentDeck].cards.length;
const known = decks[currentDeck].cards.filter((_, i) => isKnown(currentDeck, i)).length;
const percent = total ? Math.round((known / total) * 100) : 0;
document.getElementById("progressBar").style.width = percent + "%";
}

function render() {
buildDeckList();
renderCardList();
renderProgress();

document.getElementById("deckDescription").textContent = decks[currentDeck].description;

if (filteredIndexes.length === 0) {
document.getElementById("counter").textContent = "0 / 0";
document.getElementById("sideLabel").textContent = "No Cards";
document.getElementById("cardText").textContent = "Change your search or filter.";
return;
}

const realIndex = filteredIndexes[currentCard];
const card = decks[currentDeck].cards[realIndex];

document.getElementById("counter").textContent = `${currentCard + 1} / ${filteredIndexes.length}`;
document.getElementById("sideLabel").textContent = showingAnswer ? "Answer" : "Question";
document.getElementById("cardText").textContent = showingAnswer ? card[1] : card[0];
document.getElementById("cardText").className = showingAnswer ? "card-text answer" : "card-text";
document.getElementById("starBtn").textContent = isStarred(currentDeck, realIndex) ? "★ Starred" : "☆ Star";
}

function refresh() {
applyFilter();
render();
}

function flipCard() {
showingAnswer = !showingAnswer;
render();
}

function nextCard() {
if (filteredIndexes.length === 0) return;
currentCard = (currentCard + 1) % filteredIndexes.length;
showingAnswer = false;
render();
}

function prevCard() {
if (filteredIndexes.length === 0) return;
currentCard = (currentCard - 1 + filteredIndexes.length) % filteredIndexes.length;
showingAnswer = false;
render();
}

function markKnown(value) {
if (filteredIndexes.length === 0) return;
const realIndex = filteredIndexes[currentCard];
localStorage.setItem(cardKey("known", currentDeck, realIndex), value ? "true" : "false");
refresh();
}

function toggleStar() {
if (filteredIndexes.length === 0) return;
const realIndex = filteredIndexes[currentCard];
const value = !isStarred(currentDeck, realIndex);
localStorage.setItem(cardKey("starred", currentDeck, realIndex), value ? "true" : "false");
refresh();
}

function shuffleCards() {
for (let i = filteredIndexes.length - 1; i > 0; i--) {
const j = Math.floor(Math.random() * (i + 1));
[filteredIndexes[i], filteredIndexes[j]] = [filteredIndexes[j], filteredIndexes[i]];
}
currentCard = 0;
showingAnswer = false;
render();
}

function resetDeck() {
if (!confirm("Reset known/starred progress for this deck?")) return;

decks[currentDeck].cards.forEach((_, i) => {
localStorage.removeItem(cardKey("known", currentDeck, i));
localStorage.removeItem(cardKey("starred", currentDeck, i));
});

currentCard = 0;
showingAnswer = false;
refresh();
}

document.getElementById("mode").addEventListener("change", () => {
currentCard = 0;
showingAnswer = false;
refresh();
});

document.getElementById("searchBox").addEventListener("input", () => {
currentCard = 0;
showingAnswer = false;
refresh();
});

document.addEventListener("keydown", e => {
if (e.target.tagName === "INPUT" || e.target.tagName === "SELECT") return;

if (e.code === "Space") {
e.preventDefault();
flipCard();
}

if (e.key === "ArrowRight") nextCard();
if (e.key === "ArrowLeft") prevCard();
if (e.key.toLowerCase() === "k") markKnown(true);
if (e.key.toLowerCase() === "s") toggleStar();
});

refresh();
</script>
